    72→@router.get("/overview")

    75→    概览 Tab 数据
    76→
    77→    返回：
    78→    - insight: 一句话洞察
    79→    - state_score: 状态分数 + 三信号分解 + 补救动作
    80→    - perma: PERMA 五维快照
    81→    - today_tasks: 今日任务（≤3条）
    82→    - risk_alerts: 风险提示
    83→    """
    84→    auth = require_auth(request)
    85→    user_id = int(auth["user_id"])
    86→
    87→    # State Score
    88→    state_score = soul_os.calculate_state_score(user_id)
    89→
    90→    # L1 Schema (PERMA)
    91→    l1 = soul_os.get_l1_schema(user_id)
    92→    perma_data = l1.perma.to_dict()

    94→    # 今日任务
    95→    tasks = soul_os.list_commitments(user_id, status=["suggested", "active"], limit=3)
    96→
    97→    # 反依赖检查
    98→    anti_dep = soul_os.check_anti_dependency(user_id)
    99→    risk_alerts = []
   100→    if anti_dep.should_intervene:
   101→        risk_alerts.append({
   102→            "type": anti_dep.intervention_type,
   103→            "message": anti_dep.intervention_message,
   104→        })
   105→
   106→    # 生成洞察
   107→    score = state_score.score
   108→    if score >= 70:
   109→        insight = "今天状态不错，保持这个节奏！"
   110→    elif score >= 50:
   111→        insight = "状态还行，做一个小任务可以让你感觉更好。"
   112→    elif score >= 30:
   113→        insight = "今天需要一点自我关怀，先做一个最小行动。"
   114→    else:
   115→        insight = "今天可能有点累，先照顾好自己。"
   116→
   117→    return _ok({
   118→        "insight": insight,
   119→        "state_score": state_score.to_dict(),
   120→        "perma": {
   121→            "positive_emotion": perma_data["positive_emotion"],
   122→            "engagement": perma_data["engagement"],
   123→            "relationships": perma_data["relationships"],
   124→            "meaning": perma_data["meaning"],
   125→            "accomplishment": perma_data["accomplishment"],
   126→        },
   127→        "today_tasks": [
   128→            {
   129→                "task_id": t["task_id"],
   130→                "title": t["title"],
   131→                "status": t["status"],
   132→                "commitment_type": t.get("commitment_type", "start_task"),
   133→            }
   134→            for t in tasks

   136→        "risk_alerts": risk_alerts,

   188→# Tab 3: 趋势 (Trends)
   189→# =============================================================================
   190→
   191→@router.get("/trends")
   192→def get_trends(request: Request, days: int = 7):
   193→    """
   194→    趋势 Tab 数据
   195→
   196→    返回：
   197→    - state_score_history: 历史分数
   198→    - events: 关键事件（任务完成、打卡等）
   199→    """
   200→    auth = require_auth(request)
   201→    user_id = int(auth["user_id"])
   202→
   203→    days = max(1, min(30, int(days)))
   204→    window_start = datetime.now(timezone.utc) - timedelta(days=days)
   205→
   206→    # 获取打卡历史计算每日分数
   207→    checkins = fortune_db.fetch_all(
   208→        """
   209→        SELECT DATE(created_at) as date, AVG(intensity) as avg_intensity
   210→        FROM fortune_checkin
   211→        WHERE user_id = %s AND created_at >= %s
   212→        GROUP BY DATE(created_at)
   213→        ORDER BY date
   214→        """,
   215→        (user_id, window_start),
   216→    ) or []
   217→
   218→    # 获取任务完成历史
   219→    done_by_date = fortune_db.fetch_all(
   220→        """
   221→        SELECT DATE(done_at) as date, COUNT(*) as cnt
   222→        FROM fortune_commitment
   223→        WHERE user_id = %s AND status = 'done' AND done_at >= %s
   224→        GROUP BY DATE(done_at)
   225→        """,
   226→        (user_id, window_start),
   227→    ) or []
   228→    done_map = {str(r["date"]): r["cnt"] for r in done_by_date}
   229→
   230→    # 构建历史分数（简化版）
   231→    score_history = []
   232→    for c in checkins:
   233→        date_str = str(c["date"])
   234→        intensity = float(c.get("avg_intensity") or 5)
   235→        emotion_signal = max(0, (10 - intensity)) * 4
   236→        action_signal = min(done_map.get(date_str, 0) * 10, 40)
   237→        score = int(emotion_signal + action_signal + 10)  # 简化 streak
   238→        score_history.append({
   239→            "date": date_str,
   240→            "score": score,
   241→        })
   242→
   243→    # 获取事件列表
   244→    events = []
   245→
   246→    # 任务完成事件
   247→    done_tasks = fortune_db.fetch_all(
   248→        """
   249→        SELECT task_id, title, done_at
   250→        FROM fortune_commitment
   251→        WHERE user_id = %s AND status = 'done' AND done_at >= %s
   252→        ORDER BY done_at DESC
   253→        LIMIT 20
   254→        """,
   255→        (user_id, window_start),
   256→    ) or []
   257→    for t in done_tasks:
   258→        events.append({
   259→            "date": str(t["done_at"].date()) if t.get("done_at") else "",
   260→            "type": "commitment_done",
   261→            "title": f"完成了：{t.get('title', '')}",
   262→        })
   263→
   264→    # 打卡事件
   265→    recent_checkins = fortune_db.fetch_all(
   266→        """
   267→        SELECT mood, intensity, created_at
   268→        FROM fortune_checkin
   269→        WHERE user_id = %s AND created_at >= %s
   270→        ORDER BY created_at DESC
   271→        LIMIT 10
   272→        """,
   273→        (user_id, window_start),
   274→    ) or []
   275→    for c in recent_checkins:
   276→        events.append({
   277→            "date": str(c["created_at"].date()) if c.get("created_at") else "",
   278→            "type": "checkin",
   279→            "title": f"情绪打卡：{c.get('mood', '')}（{c.get('intensity', 0)}/10）",
   280→        })
   281→
   282→    # 按日期排序
   283→    events.sort(key=lambda x: x["date"], reverse=True)
   284→
   285→    return _ok({
   286→        "state_score_history": score_history,
   287→        "events": events[:20],
   288→    })

   319→@router.post("/task/accept")
   320→def accept_task(req: TaskAcceptRequest, request: Request):
   321→    """接受任务"""
   322→    auth = require_auth(request)
   323→    require_csrf(request, auth)
   324→    user_id = int(auth["user_id"])
   325→    task_id = (req.task_id or "").strip()
   326→
   327→    if not _UUID_RE.match(task_id):
   328→        return _err(400, "invalid_request", "invalid_task_id")
   329→
   330→    ctype = (req.commitment_type or "").strip() if req.commitment_type else None
   331→    if ctype is not None and ctype not in ("start_task", "schedule_task"):
   332→        return _err(400, "invalid_request", "invalid_commitment_type")
   333→
   334→    # 计算 due_at
   335→    due_at = None
   336→    if ctype == "schedule_task":
   337→        prefs = fortune_db.fetch_one(
   338→            "SELECT push_timezone FROM fortune_user_preferences WHERE user_id=%s",
   339→            (user_id,),
   340→        ) or {}
   341→        tz_name = str(prefs.get("push_timezone") or "Asia/Shanghai")
   342→        try:
   343→            tz = ZoneInfo(tz_name)
   344→        except Exception:
   345→            tz = ZoneInfo("Asia/Shanghai")
   346→        now_local = datetime.now(tz)
   347→        end_local = datetime.combine(now_local.date(), time(23, 0, 0), tzinfo=tz)
   348→        due_at = end_local.astimezone(timezone.utc)
   349→
   350→    row = fortune_db.execute_returning_one(
   351→        """
   352→        UPDATE fortune_commitment
   353→        SET
   354→          status = 'active',
   355→          accepted_at = COALESCE(accepted_at, now()),
   356→          commitment_type = COALESCE(%s, commitment_type),
   357→          due_at = COALESCE(%s, due_at)
   358→        WHERE task_id=%s AND user_id=%s AND status IN ('suggested','active')
   359→        RETURNING task_id, status, accepted_at
   360→        """,
   361→        (ctype, due_at, task_id, user_id),
   362→    )
   363→
   364→    if not row:
   365→        return _err(404, "not_found", "task_not_found")
   366→
   367→    return _ok({
   368→        "task_id": str(row["task_id"]),
   369→        "status": str(row["status"]),
   370→        "accepted_at": str(row.get("accepted_at") or ""),
   371→    })

   377→    完成任务
   378→
   379→    实现完整反馈回路：
   380→    1. 更新任务状态

   382→    3. 更新 growth_streak
   383→    4. 返回新的 State Score
   384→    """

   387→    user_id = int(auth["user_id"])

   389→

   406→            user_id=user_id,
   407→            currency_type=CurrencyType.AURA,
   408→            amount=10,
   409→            category=LedgerCategory.EARN,
   410→            reason="commitment_complete",
   411→            ref_type="commitment",
   412→            ref_id=task_id,
   413→            description="完成任务奖励",
   414→        )
   415→        aura_earned = entry.amount
   416→    except Exception:
   417→        pass
   418→
   419→    # 更新 streak
   420→    new_streak = 0
   421→    try:
   422→        from services import twin_service
   423→        new_streak = twin_service.increment_growth_streak(user_id)
   424→    except Exception:
   425→        pass
   426→
   427→    return _ok({
   428→        "task_id": task_id,
   429→        "status": "done",
   430→        "state_score": result.get("state_score", {}),
   431→        "rewards": {
   432→            "aura_earned": aura_earned,
   433→            "new_streak": new_streak,
   434→        },
   435→        "message": result.get("message", ""),

   572→@router.get("/kline")
   573→def get_kline(request: Request, days: int = 30, interval: str = "day"):
   574→    """
   575→    K线数据 API
   576→
   577→    参数：
   578→    - days: 获取天数 (1-90)
   579→    - interval: 时间间隔 (day | week | month)

   672→def _calculate_kline_from_events(user_id: int, window_start: datetime, days: int) -> List[Dict]:
   673→    """从事件数据实时计算K线OHLC"""
   674→    # 获取每日打卡数据
   675→    checkins_by_date = fortune_db.fetch_all(
   676→        """
   677→        SELECT DATE(created_at) as date,
   678→               MIN(intensity) as min_intensity,
   679→               MAX(intensity) as max_intensity,
   680→               AVG(intensity) as avg_intensity,
   681→               COUNT(*) as cnt
   682→        FROM fortune_checkin
   683→        WHERE user_id = %s AND created_at >= %s
   684→        GROUP BY DATE(created_at)
   685→        ORDER BY date
   686→        """,
   687→        (user_id, window_start),
   688→    ) or []
   689→
   690→    # 获取每日任务完成数
   691→    tasks_by_date = fortune_db.fetch_all(
   692→        """
   693→        SELECT DATE(done_at) as date, COUNT(*) as cnt
   694→        FROM fortune_commitment
   695→        WHERE user_id = %s AND status = 'done' AND done_at >= %s
   696→        GROUP BY DATE(done_at)
   697→        """,
   698→        (user_id, window_start),
   699→    ) or []
   700→    task_map = {str(r["date"]): r["cnt"] for r in tasks_by_date}
   701→
   702→    # 构建日期范围
   703→    result = []
   704→    current_date = window_start.date()
   705→    end_date = datetime.now(timezone.utc).date()
   706→
   707→    checkin_map = {str(c["date"]): c for c in checkins_by_date}
   708→
   709→    prev_close = 50.0  # 默认起始分数
   710→
   711→    while current_date <= end_date:
   712→        date_str = str(current_date)
   713→        c = checkin_map.get(date_str)
   714→        task_cnt = task_map.get(date_str, 0)
   715→
   716→        if c:
   717→            # 有打卡数据
   718→            avg_intensity = float(c.get("avg_intensity") or 5)
   719→            min_intensity = float(c.get("min_intensity") or avg_intensity)
   720→            max_intensity = float(c.get("max_intensity") or avg_intensity)
   721→
   722→            # 将情绪强度转换为分数 (intensity越低越好)
   723→            # intensity 0-10 映射到 score 100-0
   724→            base_score = max(0, 100 - avg_intensity * 10)
   725→
   726→            # 行动分数加成
   727→            action_bonus = min(task_cnt * 5, 20)
   728→
   729→            open_score = prev_close
   730→            close_score = min(100, base_score + action_bonus)
   731→            high_score = max(open_score, close_score, 100 - min_intensity * 10 + action_bonus)
   732→            low_score = min(open_score, close_score, 100 - max_intensity * 10)
   733→
   734→            drivers = []
   735→            if task_cnt > 0:
   736→                drivers.append(f"完成{task_cnt}个任务")
   737→            if c.get("cnt", 0) > 1:
   738→                drivers.append(f"{c['cnt']}次打卡")
   739→
   740→            result.append({
   741→                "date": current_date,
   742→                "open_score": round(open_score, 1),
   743→                "high_score": round(high_score, 1),
   744→                "low_score": round(low_score, 1),
   745→                "close_score": round(close_score, 1),
   746→                "volume": c.get("cnt", 0) + task_cnt,
   747→                "action_count": task_cnt,
   748→                "emotion_avg": round(avg_intensity, 1),
   749→                "drivers": drivers,
   750→            })
   751→            prev_close = close_score
   752→        else:
   753→            # 无数据日，保持前一天收盘
   754→            if task_cnt > 0:
   755→                action_bonus = min(task_cnt * 5, 20)
   756→                close_score = min(100, prev_close + action_bonus)
   757→                result.append({
   758→                    "date": current_date,
   759→                    "open_score": round(prev_close, 1),
   760→                    "high_score": round(close_score, 1),
   761→                    "low_score": round(prev_close, 1),
   762→                    "close_score": round(close_score, 1),
   763→                    "volume": task_cnt,
   764→                    "action_count": task_cnt,
   765→                    "drivers": [f"完成{task_cnt}个任务"],
   766→                })
   767→                prev_close = close_score
   768→
   769→        current_date += timedelta(days=1)
   770→
   771→    return result

   774→def _aggregate_kline(data: List[Dict], interval: str) -> List[Dict]:
   775→    """聚合K线数据到周/月"""
   776→    if not data:
   777→        return []
   778→
   779→    from collections import defaultdict
   780→
   781→    groups = defaultdict(list)
   782→
   783→    for d in data:
   784→        date_str = d["date"]
   785→        if interval == "week":
   786→            # ISO周
   787→            dt = datetime.strptime(date_str, "%Y-%m-%d")
   788→            key = f"{dt.year}-W{dt.isocalendar()[1]:02d}"
   789→        else:  # month
   790→            key = date_str[:7]  # YYYY-MM
   791→        groups[key].append(d)
   792→
   793→    result = []
   794→    for key in sorted(groups.keys()):
   795→        items = groups[key]
   796→        result.append({
   797→            "date": key,
   798→            "open": items[0]["open"],
   799→            "high": max(item["high"] for item in items),
   800→            "low": min(item["low"] for item in items),
   801→            "close": items[-1]["close"],
   802→            "volume": sum(item["volume"] for item in items),
   803→            "trend": "up" if items[-1]["close"] > items[0]["open"] else ("down" if items[-1]["close"] < items[0]["open"] else "neutral"),
   804→            "color": "#22c55e" if items[-1]["close"] > items[0]["open"] else ("#ef4444" if items[-1]["close"] < items[0]["open"] else "#f59e0b"),
   805→            "drivers": [],
   806→        })
   807→
   808→    return result
