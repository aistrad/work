    24→        WITH q AS (SELECT plainto_tsquery('simple', %s) AS ts)

    26→          d.doc_id,
    27→          d.title,
    28→          d.file_name,

    30→          c.chunk_no,

    36→        CROSS JOIN q
    37→        WHERE (c.content_tsv @@ q.ts)
    38→           OR (similarity(c.content, %s) > 0.15)
    39→        ORDER BY (ts_rank_cd(c.content_tsv, q.ts) + similarity(c.content, %s)) DESC
    40→        LIMIT %s
