     1→"use client";

     3→import { HTMLAttributes, forwardRef, memo } from "react";
     4→import { cn } from "@/lib/utils";
     5→
     6→// ═══════════════════════════════════════════════════════════════════════════
     7→// Card - v7.1 Linear 风格设计系统
     8→// 无边框设计 + 多层阴影
     9→// Vercel rule: rerender-memo
    10→// ═══════════════════════════════════════════════════════════════════════════
    11→
    12→export interface CardProps extends HTMLAttributes<HTMLDivElement> {
    13→  variant?: "default" | "flat" | "elevated" | "glass";
    14→  hover?: boolean;
    15→  padding?: "none" | "sm" | "md" | "lg";
    16→}
    17→
    18→export const Card = memo(forwardRef<HTMLDivElement, CardProps>(
    19→  ({ className, variant = "default", hover = true, padding = "md", children, ...props }, ref) => {
    20→    const variants = {
    21→      // Default: v7.1 无边框设计 + 多层阴影
    22→      default: cn(
    23→        "bg-bg-card",
    24→        "shadow-card",
    25→        hover && "hover:shadow-card-hover transition-shadow duration-normal ease-out-expo"
    26→      ),
    27→      // Flat: 纯色背景，微妙边框
    28→      flat: cn(
    29→        "bg-bg-card",
    30→        "border border-subtle/50",
    31→        hover && "hover:bg-bg-card-hover transition-colors duration-fast"
    32→      ),
    33→      // Elevated: 浮起效果
    34→      elevated: cn(
    35→        "bg-bg-elevated",
    36→        "shadow-elevated",
    37→        hover && "hover:shadow-float hover:-translate-y-0.5 transition-all duration-normal ease-out-expo"
    38→      ),
    39→      // Glass: 毛玻璃效果
    40→      glass: cn(
    41→        "bg-bg-card/80 backdrop-blur-md",
    42→        "border border-white/20",
    43→        "shadow-card",
    44→        hover && "hover:bg-bg-card/90 hover:shadow-card-hover transition-all duration-normal"
    45→      ),
    46→    };
    47→
    48→    const paddings = {
    49→      none: "",
    50→      sm: "p-3",
    51→      md: "p-4",
    52→      lg: "p-6",
    53→    };
    54→
    55→    return (
    56→      <div
    57→        ref={ref}
    58→        className={cn(
    59→          "rounded-xl",
    60→          variants[variant],
    61→          paddings[padding],
    62→          className
    63→        )}
    64→        {...props}
    65→      >
    66→        {children}
    67→      </div>
    68→    );
    69→  }
    70→));
    71→
    72→Card.displayName = "Card";
    73→
