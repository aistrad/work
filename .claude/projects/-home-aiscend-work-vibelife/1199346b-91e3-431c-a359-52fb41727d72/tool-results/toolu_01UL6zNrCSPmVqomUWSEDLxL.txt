     1→# VibeLife Dashboard 设计规格
     2→
     3→> **Version**: 2.0 | 2026-01-19
     4→> **定位**: 用户人生仪表盘 - 动态洞察与行动指引
     5→> **设计理念**: Skill 驱动 + 动态内容 + 横滑浏览 + 温暖东方美学
     6→
     7→---
     8→
     9→## 核心设计哲学
    10→
    11→```
    12→┌─────────────────────────────────────────────────────────────────────────────┐
    13→│                                                                             │
    14→│   v1.0 Dashboard: 静态展示                                                  │
    15→│   ─────────────────────────────                                             │
    16→│   展示固定数据（四维原型、成长时间线）                                        │
    17→│   用户看完就离开，缺乏动态性                                                 │
    18→│                                                                             │
    19→│   v2.0 Dashboard: 动态洞察                                                  │
    20→│   ─────────────────────────────                                             │
    21→│   "每次打开，都有新的指引、新的发现、新的行动建议"                            │
    22→│                                                                             │
    23→│   核心变化：                                                                 │
    24→│   • 静态四维原型 → 动态 Skill Card（订阅驱动）                              │
    25→│   • 固定 Growth Timeline → 可发现的新 Skill                                 │
    26→│   • Lifecoach 分散 → Lifecoach 合并为单一卡片                               │
    27→│   • 横滑浏览 Skill Cards                                                    │
    28→│                                                                             │
    29→└─────────────────────────────────────────────────────────────────────────────┘
    30→```
    31→
    32→### 设计三问
    33→
    34→1. **用户为什么要来 Dashboard？**
    35→   - 获得今日方向和行动指引
    36→   - 查看订阅 Skill 的主动推送内容
    37→   - 发现新的 Skill 能力
    38→
    39→2. **Dashboard 如何让用户感到被理解？**
    40→   - 个性化问候 + 智能名言
    41→   - Skill 基于用户数据主动计算推送
    42→   - Lifecoach 整合目标-周-日一体
    43→
    44→3. **用户离开时应该带走什么？**
    45→   - 清晰：今天最重要的事
    46→   - 洞察：来自订阅 Skill 的建议
    47→   - 好奇：想探索更多 Skill
    48→

   258→## 数据模型
   259→
   260→### DashboardDTO
   261→
   262→```typescript
   263→// src/types/dashboard.ts
   264→
   265→interface DashboardDTO {
   266→  // Meta
   267→  userId: string;
   268→  generatedAt: string;
   269→
   270→  // Layer 0: Ambient
   271→  ambient: {
   272→    greeting: string;           // "早安，小薇"
   273→    quote: {
   274→      text: string;             // 智能名言
   275→      author: string | null;
   276→    };
   277→    date: string;               // "2026年1月19日 大寒前3天"
   278→    solarTerm?: string;         // "大寒"
   279→  };
   280→
   281→  // Layer 0: Status
   282→  status: {
   283→    streak: number;             // 连续签到天数
   284→    checkedIn: boolean;         // 今日是否已签到
   285→    energy?: number;            // 能量值 (0-100)
   286→  };
   287→
   288→  // Layer 1: Lifecoach (合并卡片)
   289→  lifecoach: {
   290→    northStar?: string;         // 北极星愿景
   291→
   292→    monthlyProject?: {
   293→      name: string;
   294→      progress: number;         // 0-100
   295→      daysRemaining: number;
   296→    };
   297→
   298→    todayLevers: Array<{
   299→      id: string;
   300→      text: string;
   301→      completed: boolean;
   302→    }>;
   303→
   304→    weekRocks: Array<{
   305→      id: string;
   306→      text: string;
   307→      role: string;             // 角色
   308→      completed: boolean;
   309→    }>;
   310→  };
   311→
   312→  // Layer 2: My Skills (已订阅)
   313→  mySkills: Array<{
   314→    skillId: string;
   315→    cardId: string;
   316→    title: string;              // "荣格原型洞察"
   317→    icon: string;               // "🎭"
   318→    content: SkillCardContent;  // 动态内容
   319→    computedAt: string;
   320→    actionLabel: string;        // "与原型对话"
   321→    actionRoute: string;        // "/chat?skill=jungian"
   322→  }>;
   323→
   324→  // Layer 3: Discover (未订阅推荐)
   325→  discover: Array<{
   326→    skillId: string;
   327→    title: string;              // "塔罗"
   328→    icon: string;               // "🔮"
   329→    tagline: string;            // "每日一牌"
   330→    description: string;        // "洞察潜意识"
   331→  }>;
   332→}
   333→
   334→// Skill Card 动态内容
   335→interface SkillCardContent {
   336→  headline: string;             // "今日主导: 创造者"
   337→  insights: string[];           // ["💡 创意能量旺盛", "⚠️ 注意批评者"]
   338→  rating?: {                    // 评分型内容
   339→    overall: number;
   340→    label: string;
   341→    details?: Record<string, number>;
   342→  };
   343→  suggestions?: {
   344→    do: string[];               // 宜
   345→    avoid: string[];            // 忌
   346→  };
   347→}
   348→```

   352→```typescript
   353→// Skill 注册 Dashboard Card
   354→interface SkillCardDefinition {

   357→  title: string;                // "荣格原型洞察"
   358→  icon: string;                 // "🎭"
   359→
   360→  // 计算配置
   361→  compute_endpoint: string;     // "/api/skills/jungian/dashboard"
   362→  refresh: "daily" | "weekly" | "realtime";
   363→
   364→  // 卡片配置
   365→  size: "small" | "medium" | "large";
   366→  action_label: string;         // "与原型对话"
