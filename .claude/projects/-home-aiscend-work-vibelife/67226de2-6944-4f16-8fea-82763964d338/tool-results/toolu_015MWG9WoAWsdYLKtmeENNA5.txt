    30→class PromptBuilder:
    31→    """
    32→    System Prompt 构建器
    33→
    34→    职责：
    35→    1. 根据 Phase 构建不同的 Prompt
    36→    2. 注入会话恢复上下文
    37→    3. 注入 SOP 规则
    38→    4. 动态加载案例
    39→    5. v10.1: 用户画像注入（让 Coach 了解用户）
    40→    """
    41→
    42→    # 需要详细用户画像的 Skill（注入 goals, patterns, findings）
    43→    PORTRAIT_FULL_SKILLS = {"lifecoach", "career"}
    44→
    45→    # 需要出生信息的 Skill
    46→    BIRTH_INFO_SKILLS = {"bazi", "zodiac", "jungastro"}

    48→    def __init__(self):
    49→        self.case_index = None  # Lazy init

    51→    async def build(
    52→        self,
    53→        skill_id: Optional[str],
    54→        rule_id: Optional[str],
    55→        message: str,
    56→        profile: Optional[Dict[str, Any]] = None,
    57→        skill_data: Optional[Dict[str, Any]] = None,
    58→        session_context: Optional[SessionContext] = None,
    59→        protocol_prompt: Optional[str] = None
    60→    ) -> str:
    61→        """
    62→        构建 System Prompt
    63→
    64→        Args:
    65→            skill_id: 当前技能 ID
    66→            rule_id: 当前规则 ID
    67→            message: 用户消息
    68→            profile: 用户 Profile
    69→            skill_data: Skill 数据
    70→            session_context: 会话上下文（断点续传）
    71→            protocol_prompt: 协议专用 Prompt（可选）
    72→
    73→        Returns:
    74→            完整的 System Prompt
    75→        """
    76→        parts = []
    77→
    78→        # ═══════════════════════════════════════════════════════════════
    79→        # 协议模式：使用协议专用 Prompt
    80→        # ═══════════════════════════════════════════════════════════════
    81→        if protocol_prompt:
    82→            skill = load_skill("lifecoach")
    83→            if skill:
    84→                parts.append(f"# {skill.name}\n\n{skill.expert_persona}")
    85→            parts.append("\n---\n")
    86→            parts.append(protocol_prompt)
    87→            return "\n".join(parts)
    88→
    89→        if skill_id:
    90→            # ═══════════════════════════════════════════════════════════════
    91→            # Phase 2: Skill 执行阶段 - 完整上下文
    92→            # ═══════════════════════════════════════════════════════════════
    93→
    94→            # 基础 Skill Prompt
    95→            user_ctx = dict(profile) if profile else {}
    96→            base_prompt = build_system_prompt(skill_id, rule_id, user_ctx)
    97→            parts.append(base_prompt)
    98→
    99→            # ═══════════════════════════════════════════════════════════════
   100→            # Profile 注入（简化版 v2.2）- 两层架构
   101→            # ═══════════════════════════════════════════════════════════════
   102→            profile_context = self._build_profile_context(profile, skill_data, skill_id)
   103→            if profile_context:
   104→                parts.append(profile_context)
   105→
   106→            # 会话恢复上下文（断点续传核心）
   107→            if session_context and session_context.has_checkpoint:
   108→                resume_context = session_context.to_prompt_context()
   109→                if resume_context:
   110→                    parts.append(resume_context)
   111→
   112→            # SOP 规则
   113→            sop_rules = self._build_sop_rules(skill_id, profile, skill_data)
   114→            if sop_rules:
   115→                parts.append(sop_rules)
   116→
   117→            # 案例匹配
   118→            cases_text = await self._match_cases(skill_id, skill_data)
   119→            if cases_text:
   120→                parts.append(cases_text)
   121→
   122→        else:
   123→            # ═══════════════════════════════════════════════════════════════
   124→            # Phase 1: Skill 选择阶段 - 轻量级
   125→            # ═══════════════════════════════════════════════════════════════
   126→
   127→            core_prompt = get_phase1_prompt()
   128→            if not core_prompt:
   129→                # Fallback
   130→                core_prompt = self._get_fallback_phase1_prompt()
   131→
   132→            parts.append(core_prompt)
   133→
   134→        return "\n".join(parts)

   136→    def _build_profile_context(
   137→        self,
   138→        profile: Optional[Dict[str, Any]],
   139→        skill_data: Optional[Dict[str, Any]],
   140→        skill_id: str
   141→    ) -> str:
   142→        """
   143→        构建 Profile 上下文（v3.0 三层架构）
   144→
   145→        三层架构：
   146→        - Layer 1: Identity（共享）- 基本信息
   147→        - Layer 2: Skills（当前 Skill 专属）- skills.{skill_id}
   148→        - Layer 3: Vibe（共享深度信息）- vibe.insight + vibe.target
   149→        """
   150→        parts = []
   151→
   152→        # ═══════════════════════════════════════════════════════════════
   153→        # Layer 1: Identity（所有 Skill 共享，~200 tokens）
   154→        # ═══════════════════════════════════════════════════════════════
   155→        if profile:
   156→            identity = profile.get("identity", {})
   157→
   158→            base_info = []
   159→
   160→            # 用户名
   161→            name = identity.get("display_name")
   162→            if name:
   163→                base_info.append(f"- 用户名：{name}")
   164→
   165→            # 出生信息（如有）
   166→            birth = identity.get("birth_info", {})
   167→            if birth:
   168→                birth_date = birth.get("birth_date") or birth.get("date")
   169→                if birth_date:
   170→                    base_info.append(f"- 出生日期：{birth_date}")
   171→                birth_time = birth.get("birth_time") or birth.get("time")
   172→                if birth_time:
   173→                    base_info.append(f"- 出生时间：{birth_time}")
   174→                birth_place = birth.get("birth_place") or birth.get("location")
   175→                if birth_place:
   176→                    base_info.append(f"- 出生地点：{birth_place}")
   177→
   178→            if base_info:
   179→                parts.append("## 用户信息\n\n" + "\n".join(base_info))
   180→
   181→        # ═══════════════════════════════════════════════════════════════
   182→        # Layer 2: Skills（当前 Skill 专属，~300 tokens）
   183→        # ═══════════════════════════════════════════════════════════════
   184→        if skill_data:
   185→            # 获取该 Skill 需要的 skill_data（从 SKILL.md 配置读取）
   186→            skill_ids = get_skill_required_data(skill_id)
   187→
   188→            for sid in skill_ids:
   189→                data = skill_data.get(sid, {})
   190→                if data:
   191→                    # 过滤掉内部字段（以 _ 开头）
   192→                    filtered_data = {k: v for k, v in data.items() if not k.startswith("_")}
   193→                    if filtered_data:
   194→                        parts.append(f"\n## {sid} 数据\n\n```json\n{json.dumps(filtered_data, ensure_ascii=False, indent=2)}\n```")
   195→
   196→        # ═══════════════════════════════════════════════════════════════
   197→        # Layer 3: Vibe（共享深度信息，~200 tokens）
   198→        # ═══════════════════════════════════════════════════════════════
   199→        if profile:
   200→            vibe = profile.get("vibe", {})
   201→
   202→            # VibeInsight - 我是谁
   203→            insight = vibe.get("insight", {})
   204→            if insight:
   205→                insight_parts = []
   206→
   207→                # 本质
   208→                essence = insight.get("essence", {})
   209→                if essence:
   210→                    archetype = essence.get("archetype", {})
   211→                    if archetype.get("primary"):
   212→                        insight_parts.append(f"- 主要原型：{archetype['primary']}")
   213→                    traits = essence.get("traits", [])
   214→                    if traits:
   215→                        trait_text = ", ".join([t.get("trait", str(t)) if isinstance(t, dict) else str(t) for t in traits[:3]])
   216→                        insight_parts.append(f"- 核心特质：{trait_text}")
   217→
   218→                # 动态
   219→                dynamic = insight.get("dynamic", {})
   220→                if dynamic:
   221→                    emotion = dynamic.get("emotion", {})
   222→                    if emotion.get("current"):
   223→                        insight_parts.append(f"- 当前情绪：{emotion['current']}")
   224→                    energy = dynamic.get("energy", {})
   225→                    if energy.get("level"):
   226→                        insight_parts.append(f"- 能量水平：{energy['level']}")
   227→
   228→                # 规律
   229→                pattern = insight.get("pattern", {})
   230→                if pattern:
   231→                    insights_list = pattern.get("insights", [])
   232→                    if insights_list:
   233→                        insight_parts.append(f"- 洞察：{insights_list[0]}")
   234→
   235→                if insight_parts:
   236→                    parts.append("\n## 用户画像\n\n" + "\n".join(insight_parts))
   237→
   238→            # VibeTarget - 我要成为谁
   239→            target = vibe.get("target", {})
   240→            if target:
   241→                target_parts = []
   242→
   243→                # 北极星
   244→                north_star = target.get("north_star", {})
   245→                if north_star.get("vision_scene"):
   246→                    vision = north_star["vision_scene"]
   247→                    if len(vision) > 100:
   248→                        vision = vision[:100] + "..."
   249→                    target_parts.append(f"- 愿景：{vision}")
   250→
   251→                # 目标
   252→                goals = target.get("goals", [])
   253→                if goals:
   254→                    active_goals = [g for g in goals if g.get("status") == "in_progress"][:2]
   255→                    if active_goals:
   256→                        goals_text = ", ".join([g.get("title", "") for g in active_goals])
   257→                        target_parts.append(f"- 当前目标：{goals_text}")
   258→
   259→                # 聚焦
   260→                focus = target.get("focus", {})
   261→                if focus.get("primary"):
   262→                    target_parts.append(f"- 关注领域：{focus['primary']}")
   263→
   264→                if target_parts:
   265→                    parts.append("\n## 用户目标\n\n" + "\n".join(target_parts))
   266→
   267→        return "\n".join(parts) if parts else ""

   269→    def _build_sop_rules(
   270→        self,
   271→        skill_id: str,
   272→        profile: Optional[Dict[str, Any]],
   273→        skill_data: Optional[Dict[str, Any]]
   274→    ) -> str:
   275→        """
   276→        构建 SOP 规则（自然语言版）
   277→
   278→        根据当前状态生成教练式指导。
   279→        """
   280→        needs_birth = skill_requires_birth_info(skill_id)
   281→        needs_compute = skill_requires_compute(skill_id)
   282→
   283→        # 计算当前状态
   284→        status = self._compute_status(skill_id, profile, skill_data)
   285→
   286→        # 如果已经可以分析，使用简洁的状态提示
   287→        if status["ready_for_analysis"]:
   288→            template = get_sop_template("ready_for_analysis")
   289→            if template:
   290→                chart_summary = self._extract_chart_summary(skill_id, skill_data)
   291→                return template.format(
   292→                    chart_summary=chart_summary,
   293→                    skill_id=skill_id
   294→                )
   295→            return self._get_ready_prompt()
   296→
   297→        rules_parts = []
   298→
   299→        # P1: 需要收集信息
   300→        if needs_birth and not status["has_birth_info"]:
   301→            collect_tool = get_skill_collect_tool(skill_id) or "request_info"
   302→            template = get_sop_template("need_birth_info")
   303→            if template:
   304→                rules_parts.append(template.format(collect_tool=collect_tool))
   305→            else:
   306→                rules_parts.append(self._get_need_birth_prompt(collect_tool))
   307→
   308→        # P2: 需要计算
   309→        elif needs_compute and not status["has_chart_data"]:
   310→            compute_type = get_skill_compute_type(skill_id) or skill_id
   311→            compute_tool = get_skill_compute_tool(skill_id) or f"calculate_{compute_type}"
   312→
   313→            template = get_sop_template("need_compute")
   314→            if template:
   315→                rules_parts.append(template.format(compute_tool=compute_tool, has_chart="否"))
   316→            else:
   317→                rules_parts.append(self._get_need_compute_prompt(compute_tool))
   318→
   319→        return "\n".join(rules_parts)

   321→    def _compute_status(
   322→        self,
   323→        skill_id: str,
   324→        profile: Optional[Dict[str, Any]],
   325→        skill_data: Optional[Dict[str, Any]]
   326→    ) -> Dict[str, Any]:
   327→        """计算当前 SOP 状态"""
   328→        needs_birth = skill_requires_birth_info(skill_id)
   329→        needs_compute = skill_requires_compute(skill_id)
   330→
   331→        # 检查出生信息
   332→        has_birth = False
   333→        if profile:
   334→            identity = profile.get("identity", {})
   335→            birth_info = identity.get("birth_info", {})
   336→            has_birth = bool(birth_info.get("birth_date") or birth_info.get("date"))
   337→
   338→        # 检查命盘数据
   339→        has_chart = False
   340→        if skill_data:
   341→            compute_type = get_skill_compute_type(skill_id) or skill_id
   342→            data = skill_data.get(compute_type, {})
   343→            has_chart = bool(data.get("chart") or data.get("cards"))
   344→
   345→        return {
   346→            "skill_id": skill_id,
   347→            "needs_birth_info": needs_birth,
   348→            "has_birth_info": has_birth,
   349→            "needs_compute": needs_compute,
   350→            "has_chart_data": has_chart,
   351→            "ready_for_analysis": (not needs_birth or has_birth) and (not needs_compute or has_chart)
   352→        }

   354→    def _extract_chart_summary(
   355→        self,
   356→        skill_id: str,
   357→        skill_data: Optional[Dict[str, Any]]
   358→    ) -> str:
   359→        """从命盘数据中提取关键摘要"""
   360→        if not skill_data:
   361→            return "无命盘数据"
   362→
   363→        compute_type = get_skill_compute_type(skill_id) or skill_id
   364→        data = skill_data.get(compute_type, {})
   365→
   366→        if not data:
   367→            return "无命盘数据"
   368→
   369→        summary_parts = []
   370→
   371→        # 八字
   372→        if compute_type == "bazi" and "chart" in data:
   373→            chart = data["chart"]
   374→            if "pillars" in chart and len(chart["pillars"]) > 2:
   375→                day_pillar = chart["pillars"][2]
   376→                summary_parts.append(f"日主{day_pillar.get('day_stem', '')}{day_pillar.get('day_branch', '')}")
   377→            if "month_order" in chart:
   378→                summary_parts.append(f"月令{chart['month_order']}")
   379→
   380→        # 星盘
   381→        elif compute_type == "zodiac" and "chart" in data:
   382→            chart = data["chart"]
   383→            planets = chart.get("planets", {})
   384→            if "sun" in planets:
   385→                summary_parts.append(f"太阳{planets['sun'].get('sign', '')}")
   386→            if "moon" in planets:
   387→                summary_parts.append(f"月亮{planets['moon'].get('sign', '')}")
   388→
   389→        # 塔罗
   390→        elif compute_type == "tarot" and "cards" in data:
   391→            cards = data["cards"]
   392→            if isinstance(cards, list) and len(cards) > 0:
   393→                summary_parts.append(f"{len(cards)}张牌")
   394→
   395→        if summary_parts:
   396→            return "、".join(summary_parts)
   397→        else:
   398→            return "已生成命盘"

   400→    async def _match_cases(
   401→        self,
   402→        skill_id: str,
   403→        skill_data: Optional[Dict[str, Any]]
   404→    ) -> str:
   405→        """匹配相关案例"""
   406→        if not skill_data:
   407→            return ""
   408→
   409→        try:
   410→            from .case_index import get_case_index, extract_features
   411→
   412→            if self.case_index is None:
   413→                self.case_index = get_case_index()
   414→
   415→            features = extract_features(skill_data)
   416→            if not features:
   417→                return ""
   418→
   419→            cases = await self.case_index.get_matched_cases(skill_id, features, top_k=2)
   420→            if not cases:
   421→                return ""
   422→
   423→            cases_text = "\n## 相关案例\n\n"
   424→            for case in cases:
   425→                cases_text += f"### {case.name}\n{case.content}\n\n"
   426→
   427→            return cases_text
   428→
   429→        except Exception as e:
   430→            logger.warning(f"[PromptBuilder] Case matching failed: {e}")
   431→            return ""

   488→    def build_session_resume_prompt(
   489→        self,
   490→        session_context: SessionContext,
   491→        rule_name: Optional[str] = None
   492→    ) -> str:
   493→        """
   494→        构建会话恢复 Prompt（用于 SOP 模板）
   495→
   496→        专门用于断点续传场景
   497→        """
   498→        if not session_context or not session_context.has_checkpoint:
   499→            return ""
   500→
   501→        cp = session_context.checkpoint
   502→        session = session_context.session
   503→
   504→        lines = ["## 会话恢复模式\n"]
   505→
   506→        if cp:
   507→            lines.append(f"用户上次完成了 {cp.step} 个步骤。\n")
   508→
   509→            if cp.collected_data:
   510→                lines.append("**已收集信息**：")
   511→                for key, value in cp.collected_data.items():
   512→                    lines.append(f"- {key}: {value}")
   513→                lines.append("")
   514→
   515→            next_step = cp.step + 1
   516→            lines.append(f"**下一步**：")
   517→            lines.append(f"继续第 {next_step} 个问题\n")
   518→
   519→        lines.append("**处理方式**：")
   520→        lines.append("1. 简短问候：「欢迎回来！上次我们聊到了...」")
   521→        lines.append("2. 快速回顾（1-2 句）")
   522→        lines.append("3. 直接问下一个问题")
   523→
   524→        return "\n".join(lines)
