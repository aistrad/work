     8→

    12→
    13→interface SkillShowcaseCardProps {

    24→  size = 'medium',

    32→  return (
    33→    <div
    34→      onClick={onClick}
    35→      className={cn(
    36→        'group relative overflow-hidden rounded-xl cursor-pointer transition-all hover:scale-[1.02]',
    37→        'bg-bg-card border border-subtle shadow-card hover:shadow-card-hover',
    38→        size === 'large' ? 'p-5' : 'p-4'
    39→      )}
    40→    >
    41→      {/* 背景渐变 */}

    43→        className="absolute inset-0 opacity-5 group-hover:opacity-10 transition-opacity"

    45→          background: `linear-gradient(135deg, ${skill.color} 0%, transparent 100%)`,
    46→        }}
    47→      />
    48→
    49→      {/* 内容 */}
    50→      <div className="relative space-y-3">
    51→        {/* Header */}
    52→        <div className="flex items-start justify-between">

    70→              <span className="text-xs font-medium text-green-600">
    71→                {skill.category === 'core' ? '始终激活' : '已订阅'}
    72→              </span>

   110→              <span className="font-medium text-green-600">免费</span>
   111→            ) : trialRemaining && trialRemaining > 0 ? (
   112→              <span className="text-text-secondary">

   123→              onClick={(e) => {

   125→                onSubscribe?.();
   126→              }}
