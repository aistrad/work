     1→"use client";
     2→
     3→/**
     4→ * Dashboard Page - V8 人生仪表盘
     5→ *
     6→ * 以 LifeCoach 为核心的人生仪表盘系统
     7→ * 默认首页，展示用户的目标、进度和行动
     8→ */
     9→
    10→import { useState, useCallback } from "react";
    11→import { DashboardGrid, ContextualWhisper, type WhisperContext } from "@/components/dashboard";
    12→import { ProactiveLayer } from "@/components/proactive";
    13→
    14→export default function DashboardPage() {
    15→  const [whisperOpen, setWhisperOpen] = useState(false);
    16→  const [whisperContext, setWhisperContext] = useState<WhisperContext | null>(null);
    17→
    18→  const handleWhisperOpen = useCallback((context: string) => {
    19→    setWhisperContext(context as WhisperContext);
    20→    setWhisperOpen(true);
    21→  }, []);
    22→
    23→  const handleWhisperClose = useCallback(() => {
    24→    setWhisperOpen(false);
    25→    setWhisperContext(null);
    26→  }, []);
    27→
    28→  return (
    29→    <div className="h-full overflow-y-auto">
    30→      {/* Dashboard Grid */}
    31→      <DashboardGrid onWhisperOpen={handleWhisperOpen} />
    32→
    33→      {/* Contextual Whisper (半屏对话) */}
    34→      <ContextualWhisper
    35→        isOpen={whisperOpen}
    36→        context={whisperContext}
    37→        onClose={handleWhisperClose}
    38→      />
    39→
    40→      {/* Proactive Layer (灵动岛通知) */}
    41→      <ProactiveLayer />
    42→    </div>
    43→  );
    44→}
