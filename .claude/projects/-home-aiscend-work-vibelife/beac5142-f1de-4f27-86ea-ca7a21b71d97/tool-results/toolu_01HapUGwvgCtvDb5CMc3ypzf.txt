     1→import { NextRequest, NextResponse } from "next/server";

     3→// Internal API base used by Next.js server to call backend directly

     6→export async function GET(request: NextRequest) {

     9→    if (!auth) {
    10→      return NextResponse.json({ detail: "Unauthorized" }, { status: 401 });

    12→
    13→    const response = await fetch(`${API_INTERNAL}/api/v1/account/profile`, {
    14→      method: "GET",
    15→      headers: {
    16→        "Content-Type": "application/json",
    17→        Authorization: auth,
    18→      },
    19→      // include credentials if backend needs cookies; here we use bearer
    20→    });
    21→
