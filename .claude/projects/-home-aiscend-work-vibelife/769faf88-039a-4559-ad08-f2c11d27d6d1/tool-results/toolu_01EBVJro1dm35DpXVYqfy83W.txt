   143→@router.post("/stream")
   144→async def chat_stream(request: ChatRequest):
   145→    """
   146→    Send a message and get streaming response (SSE).
   147→
   148→    Returns Server-Sent Events with chunks of the response.
   149→    Final event contains [DONE] marker.
   150→    """
   151→    skill = parse_skill(request.skill)
   152→    voice_mode = parse_voice_mode(request.voice_mode or "warm")
   153→
   154→    # Get or create conversation
   155→    conversation_id = request.conversation_id or uuid4()
   156→
   157→    # Get services
   158→    llm = get_llm_service()
   159→    context_builder = get_context_builder()
   160→
   161→    async def generate():
   162→        try:
   163→            # Get user profile (for logged-in users)
   164→            profile = await get_user_profile(None)  # TODO: Get actual user
   165→
   166→            # Get conversation history
   167→            history = await get_conversation_history(conversation_id)
   168→
   169→            # Build context
   170→            system_prompt, messages = await context_builder.build(
   171→                skill=skill,
   172→                voice_mode=voice_mode,
   173→                current_message=request.message,
   174→                profile=profile,
   175→                history=history
   176→            )
   177→
   178→            # Save user message
   179→            await save_message(conversation_id, "user", request.message)
   180→
   181→            # Stream response
   182→            full_content = ""
   183→            async for chunk in llm.stream(messages):
   184→                full_content += chunk
   185→                # Send chunk as SSE event
   186→                yield {
   187→                    "event": "message",
   188→                    "data": json.dumps({
   189→                        "type": "chunk",
   190→                        "content": chunk
   191→                    })
   192→                }
   193→
   194→            # Save assistant message
   195→            await save_message(conversation_id, "assistant", full_content)
   196→
   197→            # Send completion event
   198→            yield {
   199→                "event": "message",
   200→                "data": json.dumps({
   201→                    "type": "done",
   202→                    "conversation_id": str(conversation_id),
   203→                    "skill": skill.value,
   204→                    "voice_mode": voice_mode.value
   205→                })
   206→            }
   207→
   208→        except Exception as e:
   209→            yield {
   210→                "event": "error",
   211→                "data": json.dumps({
   212→                    "type": "error",
   213→                    "message": str(e)
   214→                })
   215→            }
   216→
   217→    return EventSourceResponse(generate())

   578→@router.post("/upload")
   579→async def upload_file(
   580→    file: UploadFile = File(...),
   581→    context_hint: Optional[str] = None
   582→):
   583→    """
   584→    Upload a file for AI extraction.
   585→
   586→    Supported formats:
   587→    - Images (JPG, PNG): Screenshots from other apps, chat records
   588→    - Documents (PDF, DOCX, TXT): Resumes, diaries
   589→
   590→    Returns extracted information that can be merged into user profile.
   591→    """
   592→    # Read file data
   593→    file_data = await file.read()
   594→
   595→    # Check file size (10MB limit)
   596→    if len(file_data) > 10 * 1024 * 1024:
   597→        raise HTTPException(status_code=413, detail="文件过大，最大支持 10MB")
   598→
   599→    # Extract information
   600→    extraction_service = get_extraction_service()
   601→
   602→    result = await extraction_service.extract_from_file(
   603→        file_data=file_data,
   604→        filename=file.filename or "unknown",
   605→        mime_type=file.content_type,
   606→        context_hint=context_hint
   607→    )
   608→
   609→    return {
   610→        "success": result.success,
   611→        "file_type": result.file_type,
   612→        "content_type": result.content_type,
   613→        "summary": result.summary,
   614→        "confidence": result.confidence,
   615→        "extracted_data": result.extracted_data,
   616→        "profile_updates": result.profile_updates,
   617→        "error": result.error
   618→    }

   621→@router.post("/upload/batch")
   622→async def upload_files_batch(
   623→    files: List[UploadFile] = File(...)
   624→):
   625→    """
   626→    Upload multiple files for batch extraction.
   627→    """
   628→    if len(files) > 5:
   629→        raise HTTPException(status_code=400, detail="最多同时上传 5 个文件")
   630→
   631→    extraction_service = get_extraction_service()
   632→    results = []
   633→
   634→    for file in files:
   635→        file_data = await file.read()
   636→
   637→        if len(file_data) > 10 * 1024 * 1024:
   638→            results.append({
   639→                "filename": file.filename,
   640→                "success": False,
   641→                "error": "文件过大"
   642→            })
   643→            continue
   644→
   645→        result = await extraction_service.extract_from_file(
   646→            file_data=file_data,
   647→            filename=file.filename or "unknown",
   648→            mime_type=file.content_type
   649→        )
   650→
   651→        results.append({
   652→            "filename": file.filename,
   653→            "success": result.success,
   654→            "content_type": result.content_type,
   655→            "summary": result.summary,
   656→            "error": result.error
   657→        })
   658→
   659→    return {
   660→        "total": len(files),
   661→        "successful": sum(1 for r in results if r["success"]),
   662→        "results": results
   663→    }
