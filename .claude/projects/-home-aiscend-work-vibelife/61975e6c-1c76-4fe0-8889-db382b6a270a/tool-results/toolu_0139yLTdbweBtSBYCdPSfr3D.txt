    73→def parse_skill(skill_str: str) -> Skill:
    74→    """Parse skill string to enum"""
    75→    skill_lower = skill_str.lower()
    76→    if skill_lower in ("bazi", "八字"):
    77→        return Skill.BAZI
    78→    elif skill_lower in ("zodiac", "星座"):
    79→        return Skill.ZODIAC
    80→    else:
    81→        raise HTTPException(status_code=400, detail=f"Unknown skill: {skill_str}")

    85→    """Parse voice mode string to enum"""
    86→    mode_lower = mode_str.lower()
    87→    if mode_lower in ("warm", "温暖"):
    88→        return VoiceMode.WARM
    89→    elif mode_lower in ("sarcastic", "吐槽"):

   149→@router.post("/stream")
   150→async def chat_stream(
   151→    request: ChatRequest,
   152→    current_user: Optional[CurrentUser] = Depends(get_optional_user)
   153→):
   154→    """
   155→    Send a message and get streaming response (SSE).
   156→
   157→    Returns Server-Sent Events with chunks of the response.
   158→    Final event contains [DONE] marker.
   159→    """
   160→    skill = parse_skill(request.skill)
   161→    voice_mode = parse_voice_mode(request.voice_mode or "warm")
   162→
   163→    # Get or create conversation
   164→    conversation_id = request.conversation_id or uuid4()
   165→
   166→    # Get user ID if authenticated
   167→    user_id = current_user.user_id if current_user else None
   168→
   169→    # Get services
   170→    llm = get_llm_service()
   171→    context_builder = get_context_builder()
   172→
   173→    async def generate():
   174→        try:
   175→            # Get user profile (for logged-in users)
   176→            profile = await get_user_profile(user_id)
   177→
   178→            # Get conversation history
   179→            history = await get_conversation_history(conversation_id)
   180→
   181→            # Retrieve relevant knowledge chunks (RAG)
   182→            knowledge_chunks = []
   183→            try:
   184→                knowledge_chunks = await RetrievalService.search(
   185→                    query=request.message,
   186→                    skill_id=skill.value,
   187→                    top_k=3
   188→                )
   189→            except Exception as e:
   190→                import logging
   191→                logging.warning(f"Knowledge retrieval failed: {e}")
   192→
   193→            # Build context with knowledge
   194→            system_prompt, messages = await context_builder.build(
   195→                skill=skill,
   196→                voice_mode=voice_mode,
   197→                current_message=request.message,
   198→                profile=profile,
   199→                history=history,
   200→                knowledge_chunks=knowledge_chunks
   201→            )
   202→
   203→            # Save user message
   204→            await save_message(conversation_id, "user", request.message)
   205→
   206→            # Stream response
   207→            full_content = ""
   208→            async for chunk in llm.stream(messages):
   209→                full_content += chunk
   210→                # Send chunk as SSE event
   211→                yield {
   212→                    "event": "message",
   213→                    "data": json.dumps({
   214→                        "type": "chunk",
   215→                        "content": chunk
   216→                    })
   217→                }
   218→
   219→            # Save assistant message
   220→            await save_message(conversation_id, "assistant", full_content)
   221→
   222→            # Send completion event
   223→            yield {
   224→                "event": "message",
   225→                "data": json.dumps({
   226→                    "type": "done",
   227→                    "conversation_id": str(conversation_id),
   228→                    "skill": skill.value,
   229→                    "voice_mode": voice_mode.value
   230→                })
   231→            }
   232→
   233→        except Exception as e:
   234→            yield {
   235→                "event": "error",
   236→                "data": json.dumps({
   237→                    "type": "error",
   238→                    "message": str(e)
   239→                })
   240→            }
   241→
   242→    return EventSourceResponse(generate())

   245→@router.post("/", response_model=ChatResponse)
   246→async def chat(
   247→    request: ChatRequest,
   248→    current_user: Optional[CurrentUser] = Depends(get_optional_user)
   249→):
   250→    """
   251→    Send a message and get non-streaming response.
   252→    For clients that don't support SSE.
   253→    """
   254→    skill = parse_skill(request.skill)
   255→    voice_mode = parse_voice_mode(request.voice_mode or "warm")
   256→
   257→    # Get or create conversation
   258→    conversation_id = request.conversation_id or uuid4()
   259→
   260→    # Get user ID if authenticated
   261→    user_id = current_user.user_id if current_user else None
   262→
   263→    # Get services
   264→    llm = get_llm_service()
   265→    context_builder = get_context_builder()
   266→
   267→    try:
   268→        # Get user profile
   269→        profile = await get_user_profile(user_id)
   270→
   271→        # Get conversation history
   272→        history = await get_conversation_history(conversation_id)
   273→
   274→        # Retrieve relevant knowledge chunks (RAG)
   275→        knowledge_chunks = []
   276→        try:
   277→            knowledge_chunks = await RetrievalService.search(
   278→                query=request.message,
   279→                skill_id=skill.value,
   280→                top_k=3
   281→            )
   282→        except Exception as e:
   283→            import logging
   284→            logging.warning(f"Knowledge retrieval failed: {e}")
   285→
   286→        # Build context with knowledge
   287→        system_prompt, messages = await context_builder.build(
   288→            skill=skill,
   289→            voice_mode=voice_mode,
   290→            current_message=request.message,
   291→            profile=profile,
   292→            history=history,
   293→            knowledge_chunks=knowledge_chunks
   294→        )
   295→
   296→        # Save user message
   297→        await save_message(conversation_id, "user", request.message)
   298→
   299→        # Get response
   300→        response = await llm.chat(messages)
   301→
   302→        # Save assistant message
   303→        await save_message(conversation_id, "assistant", response.content)
   304→
   305→        return ChatResponse(
   306→            content=response.content,
   307→            conversation_id=conversation_id,
   308→            skill=skill.value,
   309→            voice_mode=voice_mode.value,
   310→            metadata={
   311→                "model": response.model,
   312→                "usage": response.usage
   313→            }
   314→        )
   315→
   316→    except Exception as e:
   317→        raise HTTPException(status_code=500, detail=str(e))

   537→    next_question, is_complete = await interview_service.submit_answer(
   538→        session,

   541→
   542→    response = {
   543→        "is_complete": is_complete,
   544→        "next_question": None,
   545→        "progress": interview_service.get_progress(session),
   546→        "result": None
   547→    }
   548→
   549→    if is_complete:
   550→        result = interview_service.get_result(session)
   551→        response["result"] = {
   552→            "success": result.success,
   553→            "extracted_profile": result.extracted_profile,
   554→            "summary": result.summary,
   555→            "questions_answered": result.questions_answered,
   556→            "total_questions": result.total_questions
   557→        }
   558→    elif next_question:
   559→        response["next_question"] = {
   560→            "id": next_question.id,
   561→            "text": next_question.question_text,
   562→            "type": next_question.question_type
   563→        }

   623→async def upload_file(
   624→    file: UploadFile = File(...),
   625→    context_hint: Optional[str] = None
   626→):
   627→    """
   628→    Upload a file for AI extraction.
   629→
   630→    Supported formats:
   631→    - Images (JPG, PNG): Screenshots from other apps, chat records
   632→    - Documents (PDF, DOCX, TXT): Resumes, diaries
   633→
   634→    Returns extracted information that can be merged into user profile.
   635→    """
   636→    # Read file data
   637→    file_data = await file.read()
   638→
   639→    # Check file size (10MB limit)
   640→    if len(file_data) > 10 * 1024 * 1024:
   641→        raise HTTPException(status_code=413, detail="文件过大，最大支持 10MB")
   642→
   643→    # Extract information
   644→    extraction_service = get_extraction_service()
   645→
   646→    result = await extraction_service.extract_from_file(
   647→        file_data=file_data,
   648→        filename=file.filename or "unknown",
   649→        mime_type=file.content_type,
   650→        context_hint=context_hint
   651→    )
   652→
   653→    return {
   654→        "success": result.success,
   655→        "file_type": result.file_type,
   656→        "content_type": result.content_type,
   657→        "summary": result.summary,
   658→        "confidence": result.confidence,
   659→        "extracted_data": result.extracted_data,
   660→        "profile_updates": result.profile_updates,
   661→        "error": result.error
