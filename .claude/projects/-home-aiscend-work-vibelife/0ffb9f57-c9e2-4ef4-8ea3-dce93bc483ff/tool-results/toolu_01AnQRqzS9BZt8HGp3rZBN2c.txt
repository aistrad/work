    18→// 内容段类型
    19→interface TextSegment {
    20→  type: 'text';
    21→  content: string;
    22→}
    23→
    24→interface ToolSegment {
    25→  type: 'tool';
    26→  name: string;
    27→  data: unknown;
    28→}
    29→
    30→type ContentSegment = TextSegment | ToolSegment;
    31→
    32→// 解析内容，提取 [[TOOL:...]] 标记
    33→function parseContent(content: string): ContentSegment[] {
    34→  const regex = /\[\[TOOL:(\w+):([\s\S]*?)\]\]/g;
    35→  const segments: ContentSegment[] = [];
    36→  let lastIndex = 0;
    37→
    38→  const matches = Array.from(content.matchAll(regex));
    39→  for (const match of matches) {
    40→    // 前面的文本
    41→    if (match.index !== undefined && match.index > lastIndex) {
    42→      const text = content.slice(lastIndex, match.index).trim();
    43→      if (text) {
    44→        segments.push({ type: 'text', content: text });
    45→      }
    46→    }
    47→    // tool 块
    48→    try {
    49→      segments.push({
    50→        type: 'tool',
    51→        name: match[1],
    52→        data: JSON.parse(match[2])
    53→      });
    54→    } catch {
    55→      // JSON 解析失败，作为文本处理
    56→      segments.push({ type: 'text', content: match[0] });
    57→    }
    58→    lastIndex = (match.index ?? 0) + match[0].length;
    59→  }
    60→
    61→  // 剩余文本
    62→  if (lastIndex < content.length) {
    63→    const text = content.slice(lastIndex).trim();
    64→    if (text) {
    65→      segments.push({ type: 'text', content: text });
    66→    }
    67→  }
    68→
    69→  return segments;
    70→}

    72→export interface ChatMessageProps {
    73→  role: "user" | "assistant";
    74→  content: string;
    75→  timestamp?: string;
    76→  isStreaming?: boolean;
    77→  showAvatar?: boolean;
    78→  skill?: SkillType;
    79→  /**
    80→   * 表单提交回调 - 用于处理工具卡片中的表单提交
    81→   */
    82→  onSendMessage?: (message: string) => void;
    83→}

    85→export function ChatMessage({
    86→  role,
    87→  content,
    88→  timestamp,
    89→  isStreaming,
    90→  showAvatar = true,
    91→  skill = "bazi",
    92→  onSendMessage,
    93→}: ChatMessageProps) {
    94→  const isUser = role === "user";
    95→
    96→  // 解析内容，提取工具标记
    97→  const segments = useMemo(() => {
    98→    if (isUser || !content) return [];
    99→    return parseContent(content);
   100→  }, [content, isUser]);
   101→
   102→  // 判断是否有工具卡片
   103→  const hasToolCards = segments.some(s => s.type === 'tool');
   104→
   105→  return (
   106→    <article
   107→      className={cn(
   108→        // content-visibility 优化长列表 (Vercel rule: rendering-content-visibility)
   109→        "chat-message-item",
   110→        // CSS 动画替代 Framer Motion (Vercel rule: rerender-animation-library)
   111→        "animate-slide-in-up",
   112→        "flex w-full mb-4 lg:mb-6 gap-3 lg:gap-4",
   113→        isUser ? "flex-row-reverse" : "flex-row"
   114→      )}
   115→      role={isUser ? "note" : "article"}
   116→      aria-label={isUser ? "您发送的消息" : "AI 助手的回复"}
   117→    >
   118→      {/* Avatar - PC端稍大 */}
   119→      {showAvatar && !isUser && (
   120→        <div className="flex-shrink-0">
   121→          <ExpertAvatar skill={skill as any} size="md" />
   122→        </div>
   123→      )}
   124→
   125→      {/* Message content area - PC端更宽 */}
   126→      <div className={cn("flex-1 max-w-[85%] lg:max-w-[75%]", isUser && "flex flex-col items-end")}>
   127→        {isUser ? (
   128→          // 用户消息：深色气泡
   129→          <div className="chat-bubble-user">
   130→            <p className="whitespace-pre-wrap leading-relaxed text-sm lg:text-base">{content}</p>
   131→            {timestamp && <p className="chat-timestamp opacity-70 mt-1 lg:mt-1.5">{timestamp}</p>}
   132→          </div>
   133→        ) : (
   134→          // AI 消息：解析并渲染文本+工具卡片
   135→          <>
   136→            {segments.map((segment, index) => {
   137→              if (segment.type === 'text') {
   138→                return (
   139→                  <div key={index} className="chat-bubble-assistant">
   140→                    <MarkdownRenderer
   141→                      content={segment.content}
   142→                      isStreaming={isStreaming && index === segments.length - 1}
   143→                    />
   144→                    {/* Streaming indicator - Linear 风格 */}
   145→                    {isStreaming && index === segments.length - 1 && !hasToolCards && (
   146→                      <StreamingDots />
   147→                    )}
   148→                  </div>
   149→                );
   150→              } else {
   151→                return (
   152→                  <ToolCardRenderer
   153→                    key={index}
   154→                    name={segment.name}
   155→                    data={segment.data}
   156→                    skill={skill as SkillId}
   157→                    onSendMessage={onSendMessage}
   158→                  />
   159→                );
   160→              }
   161→            })}
   162→            {/* 如果没有任何段落但有内容，显示原始内容 */}
   163→            {segments.length === 0 && content && (
   164→              <div className="chat-bubble-assistant">
   165→                <MarkdownRenderer content={content} isStreaming={isStreaming} />
   166→                {isStreaming && <StreamingDots />}
   167→              </div>
   168→            )}
   169→            {timestamp && <p className="chat-timestamp mt-1">{timestamp}</p>}
   170→          </>
   171→        )}
   172→      </div>
   173→
   174→      {/* User avatar placeholder for alignment */}
   175→      {showAvatar && isUser && (
   176→        <div className="flex-shrink-0 w-8 h-8" />
   177→      )}
   178→    </article>
   179→  );
   180→}

   197→// ToolCardRenderer - 渲染工具卡片
   198→// ═══════════════════════════════════════════════════════════════════════════
   199→
   200→interface ToolCardRendererProps {
   201→  name: string;
   202→  data: unknown;
   203→  skill: SkillId;
   204→  onSendMessage?: (message: string) => void;
   205→}
   206→
   207→// 内部工具列表 - 静默忽略，不渲染任何内容
   208→const INTERNAL_TOOLS = new Set([
   209→  "use_skill",
   210→  "search_db",
   211→  "get_context",
   212→  "update_memory",
   213→  "log_event",
   214→  "activate_skills",
   215→]);
   216→
   217→function ToolCardRenderer({ name, data, skill, onSendMessage }: ToolCardRendererProps) {
   218→  const { getCardType, getCardProps } = useToolSchema({ skillId: skill });
   219→
   220→  // 0. 内部工具静默忽略
   221→  if (INTERNAL_TOOLS.has(name)) {
   222→    return null;
   223→  }
   224→
   225→  // 1. 优先使用后端返回的 cardType
   226→  const toolData = data as Record<string, unknown> | null;
   227→  const cardType = toolData?.cardType as string | undefined
   228→    || toolData?.card_type as string | undefined
   229→    || getCardType(name);
   230→
   231→  // 2. 如果有 cardType，使用 CardRegistry 渲染
   232→  if (cardType && CardRegistry.has(cardType)) {
   233→    const cardProps = getCardProps(name);
   234→
   235→    // 为表单卡片提供提交回调
   236→    const handleFormSubmit = (formData: Record<string, any>) => {
   237→      if (onSendMessage) {
   238→        // 将表单数据格式化为消息
   239→        const message = Object.entries(formData)
   240→          .map(([key, value]) => `${key}: ${value}`)
   241→          .join('\n');
   242→        onSendMessage(message);
   243→      }
   244→    };
   245→
   246→    // 统一的 onAction 处理器（用于 ShowCard 的 form、select 等）
   247→    const handleAction = (action: string, payload?: any) => {
   248→      if (action === 'submit' && onSendMessage) {
   249→        // form 提交
   250→        const message = Object.entries(payload)
   251→          .map(([key, value]) => `${key}: ${value}`)
   252→          .join('\n');
   253→        onSendMessage(message);
   254→      } else if (action === 'select' && onSendMessage) {
   255→        // select 选择
   256→        onSendMessage(String(payload));
   257→      }
   258→      // 其他 action 可以在这里添加
   259→    };
   260→
   261→    // 处理 QuestionCard 的回答提交
   262→    const handleQuestionSubmit = (answer: string) => {
   263→      if (onSendMessage) {
   264→        onSendMessage(answer);
   265→      }
   266→    };
   267→
   268→    // 适配不同卡片类型的 props 格式
   269→    const adaptedProps = cardType === 'collect_form' ? {
   270→      // CollectFormCard 专用 props
   271→      formFields: toolData?.fields || [],
   272→      title: toolData?.title,
   273→      description: toolData?.description,
   274→      allowTextInput: toolData?.allowTextInput,
   275→      textInputPlaceholder: toolData?.textInputPlaceholder,
   276→      onSubmit: handleFormSubmit,
   277→    } : cardType === 'question_card' ? {
   278→      // QuestionCard 专用 props
   279→      data: toolData,
   280→      onSubmit: handleQuestionSubmit,  // 直接传递字符串回答
   281→    } : cardType === 'skill_recommendation' ? {
   282→      // SkillRecommendationCard 专用 props
   283→      data: toolData,
   284→      onSendMessage,  // 传递消息发送回调
   285→    } : {
   286→      // 通用卡片 props（包括 form、select 等）
   287→      data: toolData,
   288→      cardProps,
   289→      onAction: handleAction,  // ShowCard 使用 onAction
   290→      onSubmit: handleFormSubmit,  // 向后兼容
   291→      ...toolData  // 传递所有后端数据
   292→    };
   293→
   294→    return (
   295→      <ToolCardErrorBoundary toolName={name}>
   296→        <Suspense fallback={<ToolLoadingFallback />}>
   297→          <div className="mt-3">
   298→            {CardRegistry.render(cardType, adaptedProps)}
   299→          </div>
   300→        </Suspense>
   301→      </ToolCardErrorBoundary>
   302→    );
   303→  }
   304→
   305→  // 3. 回退：从 skillRegistry 获取工具定义
   306→  const toolDef = skillRegistry.getTool(name);
   307→  if (toolDef?.render) {
   308→    const context: ToolExecuteContext = { skillId: skill };
   309→    return (
   310→      <ToolCardErrorBoundary toolName={name}>
   311→        <Suspense fallback={<ToolLoadingFallback />}>
   312→          <div className="mt-3">
   313→            {toolDef.render(toolData, context)}
   314→          </div>
   315→        </Suspense>
   316→      </ToolCardErrorBoundary>
   317→    );
   318→  }
   319→
   320→  // 4. Fallback: 静默忽略未知工具（不暴露技术细节）
   321→  // 仅在开发环境输出警告
   322→  if (process.env.NODE_ENV === "development") {
   323→    console.warn(`[ToolCardRenderer] Unknown tool: ${name}`, data);
   324→  }
   325→  return null;
   326→}
