   546→### 2.6 四维节点组件
   547→
   548→```tsx
   549→// components/landing/DimensionNode.tsx
   550→
   551→'use client'
   552→
   553→import { motion } from 'framer-motion'
   554→import { useState } from 'react'
   555→
   556→interface DimensionNodeProps {
   557→  name_cn: string
   558→  name_en: string
   559→  icon: string
   560→  description: string
   561→  color: string
   562→  position: 'top' | 'left' | 'right' | 'bottom'
   563→}
   564→
   565→export function DimensionNode({
   566→  name_cn,
   567→  name_en,
   568→  icon,
   569→  description,
   570→  color,
   571→  position
   572→}: DimensionNodeProps) {
   573→  const [isHovered, setIsHovered] = useState(false)
   574→
   575→  return (
   576→    <motion.div
   577→      className="relative"
   578→      onHoverStart={() => setIsHovered(true)}
   579→      onHoverEnd={() => setIsHovered(false)}
   580→    >
   581→      {/* 节点 */}
   582→      <motion.div
   583→        className="w-20 h-20 md:w-24 md:h-24 rounded-full flex flex-col items-center justify-center cursor-pointer"
   584→        style={{
   585→          backgroundColor: isHovered ? `${color}30` : `${color}15`,
   586→          border: `2px solid ${isHovered ? color : 'transparent'}`,
   587→        }}
   588→        whileHover={{ scale: 1.1 }}
   589→        transition={{ duration: 0.2 }}
   590→      >
   591→        <span className="text-2xl mb-1">{icon}</span>
   592→        <span className="text-caption font-medium">{name_cn}</span>
   593→      </motion.div>
   594→
   595→      {/* 悬浮提示 */}
   596→      {isHovered && (
   597→        <motion.div
   598→          initial={{ opacity: 0, y: 10 }}
   599→          animate={{ opacity: 1, y: 0 }}
   600→          className="absolute z-10 w-48 p-3 bg-[rgb(var(--card))] rounded-lg shadow-lg border border-[rgb(var(--border))]"
   601→          style={{
   602→            top: position === 'top' ? 'calc(100% + 8px)' : undefined,
   603→            bottom: position === 'bottom' ? 'calc(100% + 8px)' : undefined,
   604→            left: position === 'left' ? 'calc(100% + 8px)' : '50%',
   605→            transform: position === 'left' || position === 'right' ? 'translateX(0)' : 'translateX(-50%)',

   759→}
   760→
   761→export function CTAButton({
   762→  href,
   763→  children,
   764→  variant = 'primary',
   765→  size = 'default'

   767→  const baseStyles = 'inline-flex items-center justify-center rounded-xl font-medium transition-all duration-300'

   772→  }
   773→
   774→  const sizes = {
   775→    default: 'px-6 py-3 text-body',
   776→    large: 'px-8 py-4 text-large',
   777→  }
   778→
   779→  return (
   780→    <motion.div
   781→      whileHover={{ y: -2 }}
   782→      whileTap={{ scale: 0.98 }}

   785→        href={href}

   791→  )
