    96→// 生成当日问候数据
    97→function useDailyGreeting() {
    98→  return useMemo(() => {
    99→    const now = new Date();
   100→    const hour = now.getHours();
   101→
   102→    // 基于时间的问候语
   103→    const greetings = {
   104→      morning: "早安，愿你今日能量满满",
   105→      afternoon: "午后时光，适合沉淀思考",
   106→      evening: "晚间静谧，回顾今日收获",
   107→    };
   108→
   109→    let greeting = greetings.morning;
   110→    if (hour >= 12 && hour < 18) greeting = greetings.afternoon;
   111→    else if (hour >= 18) greeting = greetings.evening;
   112→
   113→    // 节气（简化版，实际应根据日期计算）
   114→    const solarTerms = ["小寒", "大寒", "立春", "雨水", "惊蛰", "春分"];
   115→    const termIdx = Math.floor((now.getMonth() * 2 + Math.floor(now.getDate() / 15)) % 6);
   116→
   117→    return {
   118→      greeting,
   119→      solarTerm: solarTerms[termIdx],
   120→      termDescription: "万物生长",
   121→      todayTip: "今日宜专注当下，一步一脚印",
   122→    };
   123→  }, []);
   124→}

   201→export function ChatContainer({
   202→  skillId,
   203→  skill = "bazi",
   204→  conversationId,
   205→  voiceMode = "warm",
   206→  onConversationStart,
   207→}: ChatContainerProps) {
   208→  const [messages, setMessages] = useState<Message[]>([]);
   209→  const [isLoading, setIsLoading] = useState(false);
   210→  const [currentConvoId, setCurrentConvoId] = useState(conversationId);
   211→  const messagesEndRef = useRef<HTMLDivElement>(null);
   212→
   213→  // Map skillId to SkillType if not provided
   214→  const activeSkill: SkillType = skill || (skillId as SkillType) || "bazi";
   215→
   216→  const scrollToBottom = () => {
   217→    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
   218→  };
   219→
   220→  useEffect(() => {
   221→    scrollToBottom();
   222→  }, [messages]);
   223→
   224→  const handleSend = async (content: string) => {
   225→    // Add user message
   226→    const userMessage: Message = {
   227→      id: Date.now().toString(),
   228→      role: "user",
   229→      content,
   230→      timestamp: new Date().toLocaleTimeString(),
   231→    };
   232→    setMessages((prev) => [...prev, userMessage]);
   233→    setIsLoading(true);
   234→
   235→    try {
   236→      const { accessToken } = getTokens();
   237→
   238→      if (accessToken) {
   239→        const response: ChatResponse = await sendMessage({
   240→          message: content,
   241→          skill_id: skillId,
   242→          conversation_id: currentConvoId,
   243→        });
   244→
   245→        // Update conversation ID
   246→        if (!currentConvoId && response.conversation_id) {
   247→          setCurrentConvoId(response.conversation_id);
   248→          onConversationStart?.(response.conversation_id);
   249→        }
   250→
   251→        // Add assistant message with potential insight
   252→        const assistantMessage: Message = {
   253→          id: (Date.now() + 1).toString(),
   254→          role: "assistant",
   255→          content: response.content,
   256→          timestamp: new Date().toLocaleTimeString(),
   257→          insight: response.insight ? {
   258→            id: response.insight.id,
   259→            insight_type: response.insight.insight_type as InsightType,
   260→            title: response.insight.title,
   261→            content: response.insight.content,
   262→          } : undefined,
   263→        };
   264→        setMessages((prev) => [...prev, assistantMessage]);
   265→      } else {
   266→        // Guest mode: use the public endpoint for progressive disclosure
   267→        const response = await sendGuestMessage(content, skillId);
   268→        const assistantMessage: Message = {
   269→          id: (Date.now() + 1).toString(),
   270→          role: "assistant",
   271→          content: response.content,
   272→          timestamp: new Date().toLocaleTimeString(),
   273→        };
   274→        setMessages((prev) => [...prev, assistantMessage]);
   275→
   276→        if (response.suggestion) {
   277→          setMessages((prev) => [
   278→            ...prev,
   279→            {
   280→              id: (Date.now() + 2).toString(),
   281→              role: "assistant",
   282→              content: response.suggestion,
   283→              timestamp: new Date().toLocaleTimeString(),
   284→            },
   285→          ]);
   286→        }
   287→      }
   288→    } catch (error) {
   289→      console.error("Chat error:", error);
   290→      // Add error message
   291→      setMessages((prev) => [...prev, {
   292→        id: (Date.now() + 1).toString(),
   293→        role: "assistant",
   294→        content: "抱歉，出了点问题。请稍后再试。",
   295→      }]);
   296→    } finally {
   297→      setIsLoading(false);
   298→    }
   299→  };
   300→
   301→  const hasMessages = messages.length > 0;
   302→
   303→  return (
   304→    <div className="flex flex-col h-full" data-skill={activeSkill}>
   305→      {/* Messages area */}
   306→      <div className={`flex-1 overflow-y-auto relative min-h-0 ${hasMessages ? 'px-4 pt-4 pb-2' : 'flex items-end justify-center pb-4'}`}>
   307→        {!hasMessages ? (
   308→          <ChatEmptyState skill={activeSkill} onQuickPrompt={handleSend} />
   309→        ) : null}
   310→        {messages.map((message) => (
   311→          <div key={message.id}>
   312→            <ChatMessage
   313→              role={message.role}
   314→              content={message.content}
   315→              timestamp={message.timestamp}
   316→            />
   317→            {/* Show insight card if present */}
   318→            {message.insight && (
   319→              <div className="ml-12 mt-2 mb-4">
   320→                <InsightCard
   321→                  type={message.insight.insight_type}
   322→                  title={message.insight.title}
   323→                  content={message.insight.content}
   324→                  onSave={() => console.log("Save insight:", message.insight?.id)}
   325→                  onShare={() => console.log("Share insight:", message.insight?.id)}
   326→                />
   327→              </div>
   328→            )}
   329→          </div>
   330→        ))}
   331→        {isLoading && (
   332→          <ChatMessage
   333→            role="assistant"
   334→            content=""
   335→            isStreaming
   336→          />
   337→        )}
   338→        <div ref={messagesEndRef} />
   339→      </div>
   340→
   341→      {/* Input */}
   342→      <ChatInput onSend={handleSend} disabled={isLoading} />
   343→    </div>
   344→  );
   345→}
