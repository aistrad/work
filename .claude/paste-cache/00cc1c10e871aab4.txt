tools:
  # 路由
  - name: activate_skills
    type: action
    description: 激活一个或多个 Skill
    parameters:
      - name: skills
        type: array
        required: true

  # 收集
  - name: request_info
    type: collect
    description: 向用户请求信息
    card_type: collect_form
    parameters:
      - name: info_type
        type: string
        required: true
        enum: [birth, context, goals]

  - name: save_birth_info
    type: action
    description: 保存用户出生信息
    parameters:
      - name: birth_date
        type: string
        required: true
      - name: birth_time
        type: string
      - name: gender
        type: string
      - name: place
        type: string

  # 状态
  - name: read_state
    type: action
    description: 读取当前 Skill 的用户状态

  - name: write_state
    type: action
    description: 写入当前 Skill 的用户状态
    parameters:
      - name: section
        type: string
        required: true
      - name: data
        type: object
        required: true

  # 检索
  - name: search_db
    type: search
    description: 检索知识库或案例库
    parameters:
      - name: query
        type: string
        required: true
      - name: top_k
        type: integer
        default: 5

  # 展示
  - name: show_card
    type: display
    description: 展示通用卡片
    parameters:
      - name: card_type
        type: string
        required: true
        enum: [list, tree, table, form, chart, progress, custom]

  - name: show_all_skills
    type: display
    description: 展示所有可用 Skill
    card_type: skill_list

  - name: recommend_skill
    type: display
    description: 推荐 Skill 给用户
    parameters:
      - name: skill_id
        type: string
        required: true
      - name: reason
        type: string
        required: true

  # 提醒
  - name: set_reminder
    type: action
    description: 设置提醒
    parameters:
      - name: title
        type: string
        required: true
      - name: schedule
        type: string
        required: true

  - name: list_reminders
    type: action
    description: 列出用户的提醒

  - name: cancel_reminder
    type: action
    description: 取消提醒
    parameters:
      - name: reminder_id
        type: string
        required: true
