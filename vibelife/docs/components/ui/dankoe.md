  2. 后端 Tool 定义                                                                                                                               
                                                                                                                                                  
  # tools.yaml                                                                                                                                    
  - name: set_north_star                                                                                                                          
    description: 设定用户的北极星愿景                                                                                                             
    tool_type: action                                                                                                                             
    parameters:                                                                                                                                   
      - name: vision                                                                                                                              
        type: string                                                                                                                              
        required: true                                                                                                                            
      - name: anti_vision                                                                                                                         
        type: array                                                                                                                               
        required: false                                                                                                                           
      - name: keywords                                                                                                                            
        type: array                                                                                                                               
        required: false                                                                                                                           
                                                                                                                                                  
  - name: set_monthly_project                                                                                                                     
    description: 设定本月项目                                                                                                                     
    tool_type: action                                                                                                                             
    parameters:                                                                                                                                   
      - name: title                                                                                                                               
        type: string                                                                                                                              
        required: true                                                                                                                            
      - name: target                                                                                                                              
        type: number                                                                                                                              
        required: false                                                                                                                           
      - name: deadline                                                                                                                            
        type: string                                                                                                                              
        required: false                                                                                                                           
                                                                                                                                                  
  - name: add_weekly_rock                                                                                                                         
    description: 添加周大石头                                                                                                                     
    tool_type: action                                                                                                                             
    parameters:                                                                                                                                   
      - name: title                                                                                                                               
        type: string                                                                                                                              
        required: true                                                                                                                            
      - name: category                                                                                                                            
        type: string                                                                                                                              
        enum: ["work", "family", "health", "learning"]                                                                                            
                                                                                                                                                  
  - name: complete_task                                                                                                                           
    description: 完成任务（大石头或日杠杆）                                                                                                       
    tool_type: action                                                                                                                             
    parameters:                                                                                                                                   
      - name: task_id                                                                                                                             
        type: string                                                                                                                              
        required: true                                                                                                                            
      - name: output                                                                                                                              
        type: string                                                                                                                              
        required: false                                                                                                                           
                                                                                                                                                  
  - name: show_journey_card                                                                                                                       
    description: 展示目标进度卡片                                                                                                                 
    tool_type: display                                                                                                                            
    card_type: JourneyCard
